//this script logs all plays to a folder inside your foobar2000 profile folder
//this feature is optional and can be turned off
//%appdata%\foobar2000\listenbrainz
//your token will be stored inside a plain text ini file
//%appdata%\foobar2000\wsh_settings\listenbrainz.ini

var panel = new _.panel("Listenbrainz", ["custom_background"]);
var buttons = new _.buttons();
//x, y, size
var listenbrainz = new _.listenbrainz(2, 2, 32);

function on_size() {
	panel.size();
	//listenbrainz.x and listenbrainz.y can be edited here if you
	//want to change the button position eg
	//listenbrainz.x = panel.w - listenbrainz.size;
	listenbrainz.update_button();
}

function on_paint(gr) {
	panel.paint(gr);
	buttons.paint(gr);
}

function on_playback_new_track() {
	listenbrainz.playback_new_track();
}

function on_playback_time() {
	listenbrainz.playback_time();
}

function on_mouse_move(x, y) {
	buttons.move(x, y);
}

function on_mouse_leave() {
	buttons.leave();
}

function on_mouse_lbtn_up(x, y) {
	buttons.lbtn_up(x, y);
}

function on_mouse_rbtn_up(x, y) {
	return panel.rbtn_up(x, y);
}
